  // Is the entry number prime or not ???
        
        
        Scanner input = new Scanner(System.in);

        int entryNumber = input.nextInt();
        boolean prime = false;
  
            for (int i = 2; i <= entryNumber; i++) {

                if (entryNumber % i == 0) {

                    prime = true;
                    break;

                }            

            if (prime) {

                System.out.println(String.format("number (%d) is prime !!!", entryNumber));

            } else {

                System.out.println(String.format("number (%d) isn't prime !!!", entryNumber));

            }
        }
